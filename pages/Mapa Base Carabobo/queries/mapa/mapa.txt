SELECT json_build_object(
    'type', 'FeatureCollection',
    'features', COALESCE(
        json_agg(
            json_build_object(
                'type', 'Feature',
                'geometry', ST_AsGeoJSON(geom, 8)::json,
                'properties', to_jsonb(t) - 'geom'
            )
        ),
        '[]'::json
    )
) AS geojson_collection
FROM public.parroquias_carabobo AS t;
