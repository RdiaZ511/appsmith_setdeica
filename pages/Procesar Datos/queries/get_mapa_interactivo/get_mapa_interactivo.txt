WITH stats AS (
    -- Agrupamos la infraestructura según los filtros de Appsmith
    SELECT 
        alcaldia, 
        COUNT(*) as total,
        STRING_AGG(datoEspecifico, ' | ') as detalles
    FROM alcaldiaInfraestructura
    WHERE 
        ( {{!tipoDato.selectedOptionValue}} OR areaDato = {{tipoDato.selectedOptionValue}} ) AND
        ( {{!gestionPor.selectedOptionValue}} OR gestionPor = {{gestionPor.selectedOptionValue}} ) AND
        ( {{!claseDato.selectedOptionValue}} OR claseDato = {{claseDato.selectedOptionValue}} )
    GROUP BY alcaldia
)
SELECT jsonb_build_object(
    'type',     'FeatureCollection',
    'features', jsonb_agg(features.feature)
)
FROM (
  SELECT jsonb_build_object(
    'type',       'Feature',
    'geometry',   ST_AsGeoJSON(geom)::jsonb, -- 'geom' es tu columna PostGIS
    'properties', jsonb_build_object(
        'NOMBRE', municipio, -- Ajusta al nombre de tu columna en la tabla GIS
        'CODIGO', municipio, -- Debe coincidir con 'VA', 'SD', etc.
        'total',  COALESCE(s.total, 0),
        'info',   COALESCE(s.detalles, 'Sin registros')
    )
  ) AS feature
  FROM parroquias_carabobo m -- Ajusta al nombre de tu tabla de polígonos
  LEFT JOIN stats s ON m.municipio = s.alcaldia
) features;
